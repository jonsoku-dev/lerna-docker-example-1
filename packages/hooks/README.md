```bash
npm install --save @jonsoku/hooks
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from '@jonsoku/hooks'
import '@jonsoku/hooks/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
